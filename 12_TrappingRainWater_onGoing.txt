class Solution:
    def trap(self, height: List[int]) -> int:
        water = 0
        left = 0
        right = 0

        lastCol = 0

        while left < len(height)-1:
            if height[left] <= lastCol: 
                left+=1
                right = left+1
                lastCol = height[right]
            while height[right] < height[left]:
                right+=1
        
        return water
