class Solution:
    def maxArea(self, height: List[int]) -> int:
        maxArea = [0,0,0,0,0]
        if len(height) == 2:
            return min(height)


        for leftIndex, left in enumerate(height):
            for rightIndex, right in enumerate(height):
                curArea = (rightIndex - leftIndex)*min(left, right)
                if curArea > maxArea[0]:
                    maxArea = [curArea, left, right, leftIndex, rightIndex]
        print(maxArea)
        return maxArea[0]
